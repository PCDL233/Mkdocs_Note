    #include<stdio.h>
#include<stdlib.h>

typedef int dataType;
typedef int Status;

const int TRUE = 1;
const int FALSE = 0;

// sequence list定义
typedef struct{
    dataType *data;
    int length;
}SqList;


// 1. 初始化list
SqList initList(dataType data[], int length){
    SqList sq;
    sq.data = data;
    sq.length = length;
    return sq;
}

// 2. 判断表是否为空
Status isEmpty(SqList sq){
    if(sq.length == 0){
        return TRUE;
    }else{
        return FALSE;
    }
}

// 3. 销毁表
Status clearList(SqList* sq){
    sq->data = NULL;
    sq->length = 0;
    return TRUE;
}
// 4. 增 -- > 插入 位置按逻辑下标
Status insertElemBaseOfIndex(SqList *sq,int index, dataType elem){
    if(index < 1 || index > sq->length){
        return FALSE;
    }
    dataType t;
    for(int i = sq->length-1; i>=index; i--){
         t = sq->data[i];
         sq->data[i] = sq->data[i+1];
         sq->data[i+1] = t;
    }

    sq->data[index] = elem;
    sq->length += 1;
    return TRUE;
}

// 5. 删
Status deleteElemBaseOfIndex(SqList* sq,int index,dataType* e){
    if(index < 1 || index > sq->length){

        return FALSE;
    }
    *e = sq->data[index-1];
    dataType t;
    for(int i=index-1;i<sq->length;i++){
        t = sq->data[i];
        sq->data[i] = sq->data[i+1];
        sq->data[i+1] = t;
    }
    sq->length--;
    return TRUE;
}

// 6. 改
Status updateElem(SqList* sq, int index, dataType e){
    if(index < 1 || index > sq->length){
        return FALSE;
    }
    sq->data[index-1] = e;
    return TRUE;
}

// 7. 查
Status searchElem(SqList* sq,dataType e){
    for(int i=0;i<sq->length;i++){
        if(sq->data[i]==e){
            return i+1;
        }
    }
    return -1;
}

// 8. 遍历
void displayList(SqList sq){
    for(int i=0;i<sq.length;i++){
        printf("%d ",sq.data[i]);
    }
    printf("\n");
}

int main(){
    system("chcp 65001");
    dataType arrs[] = {5,6,4,1,3};
    SqList  sq = initList(arrs,5);
    printf("length: %d\n",sq.length);
    displayList(sq);
    insertElemBaseOfIndex(&sq,3,0);
    printf("length: %d\n",sq.length);
    displayList(sq);
    dataType e;
    deleteElemBaseOfIndex(&sq,1,&e);
    displayList(sq);
    printf("delete elem:%d\n",e);
    updateElem(&sq,3,5);
    displayList(sq);
    int index = searchElem(&sq,5);
    printf("%d 's index: %d\n",5,index);
    clearList(&sq);
    if(isEmpty(sq)){
        printf("已销毁");
    }else{
        printf("销毁失败");
    }
    return 0;
}
