## 顺序表实现：
>### 纯手工手写
```c++
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
const int MAX = 45;
typedef int Status;
const int TRUE = 1;
const int FALSE = 0;
//学生信息数据类型定义
typedef struct
{
	char name[10];
	char num[10];
	long long int telephone;
}Student;
//顺序表定义
typedef struct
{
	Student *infor;
	int length;
}Sqlist,*SqList;

void fun()
{
	system("pause");
	system("cls");
}
//初始化
Status InitList(SqList &list)
{
	list = (SqList)malloc(sizeof(Sqlist));
	if(!list)
	{
		return FALSE;
	}
	list->infor = (Student*)malloc(MAX * sizeof(Student));
	if(list->infor ==NULL)
	{
		return FALSE;
	}
	list->length=0;
	return TRUE;
}
//赋值
void CreatList(SqList &list,Student stu[],int length)
{
	for(int i=0;i<length;i++)
	{
		list->infor[i]=stu[i];
	}
	list->length=length;
}
//删除
Status DeleteList(SqList &list)
{
	char str[10];
	printf("请输入你要删除的学生姓名：\n");
	scanf("%s",str);
	int i=0;
	while(strcmp(list->infor[i].name,str)!=0)
	{
		i++;
	}
	for(int j=i;j<=list->length;j++)
	{
		//后移
		list->infor[j]=list->infor[j+1];
	}
	list->length--;
	fun();
	return TRUE;
}
//插入
Status InsertList(SqList &list,Student e)
{
	int postion=0;
	printf("请输入你要插入的位置：\n");
	scanf("%d",&postion);
	if(postion<1||postion>list->length+1)
	{
		return FALSE;
	}
	if(list->length==MAX)
	{
		printf("顺序表已满！\n");
		return FALSE;
	}
	for(int j=list->length-1;j>=postion-1;j--)
	{
		list->infor[j+1]=list->infor[j];
	}
	list->infor[postion-1]=e;
	list->length++;
	fun();
	return TRUE;
}
//查找
int Search(SqList list,char num[])
{
	for(int i=0;i< list->length;i++)
	{
		if(strcmp(list->infor[i].num,num))
		{
			//返回序号
			return (i+1);
		}
		else
			return FALSE;
	}
	fun();
}

//修改
Status Update(SqList &list)
{
	char num[10];
	printf("请输入要修改的学生学号:\n");
	scanf("%s",num);
	Student e;
	printf("请输入修改后的学生信息：\n");
	scanf("%s%s%lld",e.name,e.num,&e.telephone);
	int i=Search(list,num);
	list->infor[i]=e;
	fun();
	return TRUE;
}

//输出
void Dispaly(SqList &list)
{
	for(int i=0;i< list->length;i++)
	{
		printf("%s\t%s\t%lld\n",list->infor[i].name,list->infor[i].num,list->infor[i].telephone);
	}
	printf("\n学生信息输出成功\a\n");
	fun();
}
//保存
Status SaveList(SqList &list,int n)
{
	FILE* sp;
	if((sp=fopen("Student.txt","w"))==NULL)
	{
		printf("打开文件错误。\n");
		exit(1);
	}
	for(int i=0;i<n;i++)
	{
		fprintf(sp,"%s\t%s\t%lld\n",list->infor[i].name,list->infor[i].num,list->infor[i].telephone);		
	}
	fclose(sp);
	printf("保存成功！\n");
	fun();
}

//录入
void Input(int &num,Student infor[])
{
	printf("请输入人数：\n");
	scanf("%d",&num);
	printf("请输入学生姓名、学号和手机号码:\n");
	for(int i=0;i<num;i++)
	{
		scanf("%s%s%lld",infor[i].name,infor[i].num,&infor[i].telephone);
	}
	fun();
}
	void menu()
{
	printf("\n"
		"*********************************************************\n"
		"*\t\t1、录入\t\t\t\t\t*\n"
		"*\t\t2、输出\t\t\t\t\t*\n"
		"*\t\t3、添加\t\t\t\t\t*\n"
		"*\t\t4、删除\t\t\t\t\t*\n"
		"*\t\t5、修改\t\t\t\t\t*\n"
		"*\t\t6、查找\t\t\t\t\t*\n"
		"*\t\t7、保存\t\t\t\t\t*\n"
		"*\t\t0、退出\t\t\t\t\t*\n"
		"*********************************************************\n"
		);
}

int main()
{
	Student infor[MAX];
	int num;	
	SqList list;
	InitList(list);
	while(1)
	{
		menu();
		int x;
		printf("请输入要执行的功能：\n");
		scanf("%d",&x);
		switch(x)
		{
		case 0: 
			{
				exit(1);
			}
		case 1:
			{
				Input(num,infor);
				CreatList(list,infor,num);
				break;
			}
		case 2:
			{
				Dispaly(list);	
				break;
			}
		case 3:
			{
				Student e;
				printf("输入要添加的学生信息：\n");
				scanf("%s%s%lld",e.name,e.num,&e.telephone);
				int y=InsertList(list,e);
				if(y==1)
					printf("添加成功！\n");
				break;
			}
		case 4:
			{
				DeleteList(list);
				break;
			}
		case 5:
			{
				Update(list);
				break;
			}
		case 6:
			{
				char number[10];
				printf("请输入要查找的学号：\n");
				scanf("%s",number);
				int c= Search(list,number);
				printf("序号：%d\n",c+1);
				break;
			}
		case 7:
			{
				SaveList(list,num);
				break;
			}
		}
	}
	return 0;
}
```

```c++
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAXSIZE 100   // 定义最大存储空间

// 学生信息结构体
typedef struct student {
    int id;            // 学号
    char name[20];     // 姓名
    char phone[20];    // 电话
} Student;

// 顺序表结构体
typedef struct {
    Student data[MAXSIZE];    // 存储空间
    int length;               // 当前长度
} SqList;

// 初始化顺序表
void InitList(SqList *L) {
    L->length = 0;
}

// 判断顺序表是否为空
int ListIsEmpty(SqList L) {
    if (L.length == 0) {
        return 1;  // 为空
    }
    else {
        return 0;  // 不为空
    }
}

// 向顺序表中插入学生信息
int InsertList(SqList *L, Student stu) {
    // 判断顺序表是否已满
    if (L->length >= MAXSIZE) {
        return 0;  // 插入失败
    }
    // 插入学生信息
    L->data[L->length] = stu;
    L->length++;
    return 1;  // 插入成功
}

// 输出顺序表中的学生信息
void PrintList(SqList L) {
    int i;
    for (i = 0; i < L.length; i++) {
        printf("学号：%d\t姓名：%s\t电话：%s\n", L.data[i].id, L.data[i].name, L.data[i].phone);
    }
}

int main() {
    SqList L;      // 声明一个顺序表
    Student stu;   // 声明一个学生信息结构体
    int i, n;      // i用于循环，n用于存储学生人数

    // 初始化顺序表
    InitList(&L);

    // 输入学生信息
    printf("请输入学生人数：");
    scanf("%d", &n);
    printf("请按照“学号 姓名 电话”的格式输入学生信息：\n");
    for (i = 0; i < n; i++) {
        scanf("%d%s%s", &stu.id, stu.name, stu.phone);
        InsertList(&L, stu);
    }

    // 判断顺序表是否为空
    if (ListIsEmpty(L)) {
        printf("顺序表为空。\n");
    }
    else {
        printf("顺序表不为空，班级通讯录信息如下：\n");
        // 输出顺序表中的学生信息
        PrintList(L);
    }

    return 0;
}
```

```c++
}
```
